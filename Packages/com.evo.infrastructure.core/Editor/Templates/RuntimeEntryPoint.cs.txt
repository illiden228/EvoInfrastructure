using System.Collections.Generic;
using System.Threading;
using Evo.Infrastructure.Runtime.Config;
using Evo.Infrastructure.Runtime.Loading;
using Evo.Infrastructure.AddressablesExtension;
using Evo.Infrastructure.Services.Config;
using Cysharp.Threading.Tasks;
using UnityEngine.SceneManagement;
using VContainer.Unity;

namespace Game.Runtime.EntryPoint
{
    public sealed class RuntimeEntryPoint : IAsyncStartable
    {
        private readonly IReadOnlyList<ILoadingStep> _steps;
        private readonly IApplicationStartupLoadingPipeline _startupLoadingPipeline;
        private readonly IConfigService _configService;

        public RuntimeEntryPoint(
            IReadOnlyList<ILoadingStep> steps,
            IApplicationStartupLoadingPipeline startupLoadingPipeline,
            IConfigService configService)
        {
            _steps = steps;
            _startupLoadingPipeline = startupLoadingPipeline;
            _configService = configService;
        }

        public async UniTask StartAsync(CancellationToken cancellationToken)
        {
            var startupScene = GetStartupScene();
            if (_startupLoadingPipeline != null)
            {
                await _startupLoadingPipeline.LoadStartupAsync(
                    _steps,
                    startupScene,
                    LoadSceneMode.Single,
                    cancellationToken: cancellationToken);
            }
        }

        private AssetReferenceScene GetStartupScene()
        {
            if (_configService != null && _configService.TryGet<ProjectConfig>(out var config))
            {
                return config?.StartupScene;
            }

            return null;
        }
    }
}
