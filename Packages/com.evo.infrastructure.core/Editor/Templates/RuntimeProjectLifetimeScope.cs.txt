using System;
using System.Reflection;
using Evo.Infrastructure.DI;
using Evo.Infrastructure.Runtime.Loading;
using Evo.Infrastructure.Runtime.UI;
using Evo.Infrastructure.Services.Ads;
using Evo.Infrastructure.Services.Analytics;
using Evo.Infrastructure.Services.Audio;
using Evo.Infrastructure.Services.Config;
using Evo.Infrastructure.Services.Focus;
using Evo.Infrastructure.Services.Leaderboard;
using Evo.Infrastructure.Services.Localization;
using Evo.Infrastructure.Services.PlatformInfo;
using Evo.Infrastructure.Services.ResourceCatalog;
using Evo.Infrastructure.Services.ResourceProvider;
using Evo.Infrastructure.Services.Save;
using Evo.Infrastructure.Services.SceneLoader;
using Evo.Infrastructure.Services.UI;
using UnityEngine;
using VContainer;
using VContainer.Unity;

namespace Game.Runtime.EntryPoint
{
    public sealed class RuntimeProjectLifetimeScope : LifetimeScope
    {
        [SerializeField] private ResourceCatalog resourceCatalog;
        [SerializeField] private ScriptableConfigCatalog[] configCatalogs;
        [SerializeField] private UiSystemConfig uiSystemConfig;

        protected override void Awake()
        {
            base.Awake();
            DontDestroyOnLoad(gameObject);
        }

        protected override void Configure(IContainerBuilder builder)
        {
            builder.RegisterEvoFeatures(features =>
            {
                features.UseConfig(configCatalogs);
                features.UsePlatform();
                features.UseFocus();
                features.UseResources(resourceCatalog);
                features.UseScene(new SceneLoaderOptions());
                features.UseLocalization();
                features.UseAudio();
                features.UseAds();
                features.UseAnalytics();
                features.UseLeaderboards();
                features.UseSave(new SaveStorageOptions());
                features.UseUi(uiSystemConfig);
                features.UseLoading(
                    new SceneTransitionOptions(),
                    new LoadingExecutionOptions
                    {
                        StepTimeoutSeconds = 45f,
                        OperationTimeoutSeconds = 180f,
                        PresentationTimeoutSeconds = 15f,
                        TransitionTimeoutSeconds = 30f,
                        OperationRetryCount = 1,
                        RetryDelaySeconds = 0.75f,
                        StepOrderMode = LoadingStepOrderMode.Registration
                    });

                RegisterOptionalYandex(features);
                RegisterOptionalCrazyGames(features);
            });

            builder.Register<Game.Runtime.Loading.LoadingViewModel>(Lifetime.Transient);
            builder.RegisterEntryPoint<Game.Runtime.Loading.LoadingViewSystem>(Lifetime.Singleton);
            builder.RegisterEntryPoint<RuntimeEntryPoint>();
        }

        private void RegisterOptionalYandex(EvoFeatureRegistry features)
        {
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.Yandex.YandexPlatformFeatureExtensions", "UseYandexPlatform");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.Yandex.YandexAdsFeatureExtensions", "UseYandexAds");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.Yandex.YandexAnalyticsFeatureExtensions", "UseYandexAnalytics");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.Yandex.YandexSaveFeatureExtensions", "UseYandexSave");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.Yandex.YandexLeaderboardsFeatureExtensions", "UseYandexLeaderboards");
        }

        private void RegisterOptionalCrazyGames(EvoFeatureRegistry features)
        {
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.CrazyGames.CrazyGamesPlatformFeatureExtensions", "UseCrazyGamesPlatform");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.CrazyGames.CrazyGamesAdsFeatureExtensions", "UseCrazyGamesAds");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.CrazyGames.CrazyGamesSaveFeatureExtensions", "UseCrazyGamesSave");
            RegisterOptionalFeature(features, "Evo.Infrastructure.Services.CrazyGames.CrazyGamesLeaderboardsFeatureExtensions", "UseCrazyGamesLeaderboards");
        }

        private void RegisterOptionalFeature(EvoFeatureRegistry features, string extensionsTypeName, string methodName)
        {
            var extensionsType = FindTypeByName(extensionsTypeName);
            var method = extensionsType?.GetMethod(methodName, BindingFlags.Public | BindingFlags.Static);
            if (method == null)
            {
                return;
            }

            method.Invoke(null, new object[] { features });
        }

        private static Type FindTypeByName(string fullName)
        {
            if (string.IsNullOrWhiteSpace(fullName))
            {
                return null;
            }

            var assemblies = AppDomain.CurrentDomain.GetAssemblies();
            for (var i = 0; i < assemblies.Length; i++)
            {
                var type = assemblies[i].GetType(fullName, false);
                if (type != null)
                {
                    return type;
                }
            }

            return null;
        }
    }
}
