using System;
using System.Collections.Generic;
using System.Reflection;
using Evo.Infrastructure.Runtime.Loading;
using Evo.Infrastructure.Runtime.UI;
using Evo.Infrastructure.Services.Ads;
using Evo.Infrastructure.Services.Analytics;
using Evo.Infrastructure.Services.Audio;
using Evo.Infrastructure.Services.Config;
using Evo.Infrastructure.Services.Focus;
using Evo.Infrastructure.Services.Leaderboard;
using Evo.Infrastructure.Services.Localization;
using Evo.Infrastructure.Services.PlatformInfo;
using Evo.Infrastructure.Services.ResourceCatalog;
using Evo.Infrastructure.Services.ResourceLoader;
using Evo.Infrastructure.Services.ResourceProvider;
using Evo.Infrastructure.Services.Save;
using Evo.Infrastructure.Services.SceneLoader;
using Evo.Infrastructure.Services.ScenePayload;
using Evo.Infrastructure.Services.UI;
using UnityEngine;
using VContainer;
using VContainer.Unity;

namespace Game.Runtime.EntryPoint
{
    public sealed class RuntimeProjectLifetimeScope : LifetimeScope
    {
        [SerializeField] private ResourceCatalog resourceCatalog;
        [SerializeField] private ScriptableConfigCatalog[] configCatalogs;
        [SerializeField] private UiSystemConfig uiSystemConfig;

        protected override void Awake()
        {
            base.Awake();
            DontDestroyOnLoad(gameObject);
        }

        protected override void Configure(IContainerBuilder builder)
        {
            if (resourceCatalog != null)
            {
                builder.RegisterInstance<IResourceCatalog>(resourceCatalog);
            }

            builder.RegisterInstance<IReadOnlyList<ScriptableConfigCatalog>>(configCatalogs ?? Array.Empty<ScriptableConfigCatalog>());
            builder.Register<IConfigProvider, ScriptableObjectConfigProvider>(Lifetime.Singleton);
            builder.Register<IConfigService, ConfigService>(Lifetime.Singleton);

            builder.Register<IPlatformInfoProvider, UnityPlatformInfoProvider>(Lifetime.Singleton);
            builder.Register<IPlatformInfoService, PlatformInfoService>(Lifetime.Singleton);
            builder.Register<IFocusService, FocusService>(Lifetime.Singleton);

            builder.Register<IResourceLoaderService, AddressablesResourceLoaderService>(Lifetime.Singleton);
            builder.RegisterInstance(new SceneLoaderOptions());
            builder.RegisterInstance(new SceneTransitionOptions());
            builder.Register<ISceneLoaderService, SceneLoaderService>(Lifetime.Singleton);
            builder.Register<IResourceProviderService, ResourceProviderService>(Lifetime.Singleton);
            builder.Register<IScenePayloadService, ScenePayloadService>(Lifetime.Singleton);
            builder.Register<ILocalizationService, LocalizationService>(Lifetime.Singleton);
            builder.Register<IAudioService, AudioService>(Lifetime.Singleton);
            builder.Register<IAdsService, AdsService>(Lifetime.Singleton);
            builder.Register<IAnalyticsService, AnalyticsService>(Lifetime.Singleton);
            builder.Register<ILeaderboardService, LeaderboardService>(Lifetime.Singleton);
            builder.Register<ISaveService, SaveService>(Lifetime.Singleton);
            builder.RegisterInstance(new SaveStorageOptions());
            builder.Register<ISaveBackend, FileSaveBackend>(Lifetime.Singleton);

            if (uiSystemConfig != null)
            {
                builder.RegisterInstance(uiSystemConfig);
                RegisterUiViewModels(builder, uiSystemConfig);
            }

            RegisterOptionalYandex(builder);

            builder.Register<IUiService, UiService>(Lifetime.Singleton);
            builder.Register<ILoadingProgress, LoadingProgressReporter>(Lifetime.Singleton);
            builder.Register<ISceneLoadingPipeline, SceneLoadingPipeline>(Lifetime.Singleton);
            builder.Register<Game.Runtime.Loading.LoadingViewModel>(Lifetime.Transient);
            builder.RegisterEntryPoint<Game.Runtime.Loading.LoadingViewSystem>(Lifetime.Singleton);
            builder.Register<ILoadingStep, TargetFrameRateStep>(Lifetime.Singleton);
            builder.RegisterEntryPoint<RuntimeEntryPoint>();
        }

        private static void RegisterUiViewModels(IContainerBuilder builder, UiSystemConfig config)
        {
            if (builder == null || config == null || config.Views == null)
            {
                return;
            }

            var registered = new HashSet<Type>();
            for (var i = 0; i < config.Views.Count; i++)
            {
                var entry = config.Views[i];
                var viewModelType = entry?.GetViewModelType();
                if (viewModelType == null || registered.Contains(viewModelType))
                {
                    continue;
                }

                builder.Register(viewModelType, Lifetime.Transient);
                registered.Add(viewModelType);
            }
        }

        private void RegisterOptionalYandex(IContainerBuilder builder)
        {
            var installerType = FindTypeByName("Evo.Infrastructure.Services.Yandex.YandexRuntimeInstaller");
            var registerMethod = installerType?.GetMethod("Register", BindingFlags.Public | BindingFlags.Static);
            if (registerMethod == null)
            {
                return;
            }

            var parameters = registerMethod.GetParameters();
            if (parameters.Length >= 2)
            {
                var configProvider = new ScriptableObjectConfigProvider(configCatalogs ?? Array.Empty<ScriptableConfigCatalog>());
                var configService = new ConfigService(new IConfigProvider[] { configProvider });
                registerMethod.Invoke(null, new object[] { builder, configService });
                return;
            }

            registerMethod.Invoke(null, new object[] { builder });
        }

        private static Type FindTypeByName(string fullName)
        {
            if (string.IsNullOrWhiteSpace(fullName))
            {
                return null;
            }

            var assemblies = AppDomain.CurrentDomain.GetAssemblies();
            for (var i = 0; i < assemblies.Length; i++)
            {
                var type = assemblies[i].GetType(fullName, false);
                if (type != null)
                {
                    return type;
                }
            }

            return null;
        }
    }
}
