using System;
using System.Threading;
using Cysharp.Threading.Tasks;
using Evo.Infrastructure.Runtime.Config;
using Evo.Infrastructure.Runtime.Loading;
using Evo.Infrastructure.Services.Config;
using Evo.Infrastructure.Services.Debug;
using Evo.Infrastructure.Services.ResourceLoader;
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using UnityEngine.ResourceManagement.ResourceProviders;
using UnityEngine.SceneManagement;
using VContainer.Unity;

namespace Game.Runtime.Loading
{
    public sealed class LoadingViewSystem : IInitializable, IDisposable, ILoadingPresentation
    {
        private const float MIN_VISIBLE_SECONDS = 0.35f;
        private const float FINAL_PROGRESS_HOLD_SECONDS = 0.1f;

        private readonly IResourceLoaderService _resourceLoader;
        private readonly ILoadingProgress _loadingProgress;
        private readonly AssetReference _loadingScene;
        private readonly LoadingExecutionOptions _loadingOptions;
        private readonly SemaphoreSlim _transitionLock = new(1, 1);
        private AsyncOperationHandle<SceneInstance> _loadingHandle;
        private bool _hasLoadingHandle;
        private double _showStartedAt;
        private LoadingScreenView _loadingScreenView;

        public bool IsVisible { get; private set; }

        public LoadingViewSystem(
            IResourceLoaderService resourceLoader,
            IConfigService configService,
            ILoadingProgress loadingProgress,
            LoadingExecutionOptions loadingOptions)
        {
            _resourceLoader = resourceLoader;
            _loadingProgress = loadingProgress;
            _loadingOptions = loadingOptions ?? new LoadingExecutionOptions();

            if (configService != null && configService.TryGet<ProjectConfig>(out var config))
            {
                _loadingScene = config?.LoadingScene;
            }
        }

        public void Initialize()
        {
            if (_loadingProgress == null)
            {
                EvoDebug.LogWarning("LoadingSceneSystem skipped: ILoadingProgress is null.", nameof(LoadingViewSystem));
                return;
            }

            _loadingProgress.Ready += OnReady;
            _loadingProgress.Finished += OnFinished;
        }

        public void Dispose()
        {
            if (_loadingProgress != null)
            {
                _loadingProgress.Ready -= OnReady;
                _loadingProgress.Finished -= OnFinished;
            }

            _transitionLock.Dispose();
        }

        private void OnReady()
        {
            ShowAsync(CancellationToken.None).Forget();
        }

        private void OnFinished()
        {
            HideAsync(CancellationToken.None).Forget();
        }

        public UniTask ShowAsync(CancellationToken cancellationToken)
        {
            return ShowLoadingSceneAsync(cancellationToken);
        }

        public UniTask HideAsync(CancellationToken cancellationToken)
        {
            return HideLoadingSceneAsync(cancellationToken);
        }

        private async UniTask ShowLoadingSceneAsync(CancellationToken cancellationToken = default)
        {
            if (_loadingScene == null)
            {
                return;
            }

            using var timeout = CreatePresentationTimeout(cancellationToken);
            var operationToken = timeout.Token;
            var lockTaken = false;
            try
            {
                await _transitionLock.WaitAsync(operationToken);
                lockTaken = true;
                if (IsVisible)
                {
                    return;
                }

                _showStartedAt = Time.realtimeSinceStartupAsDouble;
                EvoDebug.Log("LoadingSceneSystem loading LoadingScene.", nameof(LoadingViewSystem));
                if (!_hasLoadingHandle || !_loadingHandle.IsValid())
                {
                    _loadingHandle = _resourceLoader.LoadSceneHandle(_loadingScene, LoadSceneMode.Additive, true, 100);
                    _hasLoadingHandle = _loadingHandle.IsValid();
                }

                if (_hasLoadingHandle)
                {
                    await _loadingHandle.Task.AsUniTask().AttachExternalCancellation(operationToken);
                    _loadingScreenView = FindLoadingView(_loadingHandle.Result.Scene);
                    if (_loadingScreenView == null)
                    {
                        throw new InvalidOperationException("LoadingScene does not contain LoadingScreenView.");
                    }

                    await _loadingScreenView.PlayShowAsync(operationToken);
                    await UniTask.Yield(PlayerLoopTiming.Update);
                }
                else
                {
                    EvoDebug.LogWarning(
                        "LoadingScene handle is invalid. Target loading will continue without presentation.",
                        nameof(LoadingViewSystem));
                    return;
                }

                IsVisible = true;
            }
            catch (OperationCanceledException)
            {
                ResetLoadingHandle(release: true);
                if (cancellationToken.IsCancellationRequested)
                {
                    throw;
                }

                EvoDebug.LogWarning(
                    $"Loading presentation timed out after {_loadingOptions.PresentationTimeoutSeconds:0.###} seconds. Target loading will continue without it.",
                    nameof(LoadingViewSystem));
            }
            catch (Exception ex)
            {
                ResetLoadingHandle(release: true);
                EvoDebug.LogError($"LoadingSceneSystem failed to load LoadingScene. {ex.Message}", nameof(LoadingViewSystem));
            }
            finally
            {
                if (lockTaken)
                {
                    _transitionLock.Release();
                }
            }
        }

        private async UniTask HideLoadingSceneAsync(CancellationToken cancellationToken)
        {
            if (_loadingScene == null)
            {
                return;
            }

            using var timeout = CreatePresentationTimeout(cancellationToken);
            var operationToken = timeout.Token;
            var lockTaken = false;
            try
            {
                await _transitionLock.WaitAsync(operationToken);
                lockTaken = true;
                if (!IsVisible && !(_hasLoadingHandle && _loadingHandle.IsValid()))
                {
                    return;
                }

                _loadingProgress?.Report(1f, "Finalizing", 0, 0);
                await WaitMinimumVisibleDurationAsync(operationToken);
                await UniTask.Delay(
                    TimeSpan.FromSeconds(FINAL_PROGRESS_HOLD_SECONDS),
                    DelayType.UnscaledDeltaTime,
                    PlayerLoopTiming.Update,
                    operationToken);

                if (_loadingScreenView != null)
                {
                    await _loadingScreenView.PlayHideAsync(operationToken);
                }

                EvoDebug.Log("LoadingSceneSystem unloading LoadingScene.", nameof(LoadingViewSystem));
                if (_hasLoadingHandle && _loadingHandle.IsValid())
                {
                    await _resourceLoader.UnloadSceneAsync(_loadingScene, operationToken);
                }
                else
                {
                    await _resourceLoader.UnloadSceneAsync(_loadingScene, operationToken);
                }
            }
            catch (OperationCanceledException)
            {
                ResetLoadingHandle(release: true);
                if (cancellationToken.IsCancellationRequested)
                {
                    throw;
                }

                EvoDebug.LogWarning(
                    $"Loading presentation hide timed out after {_loadingOptions.PresentationTimeoutSeconds:0.###} seconds.",
                    nameof(LoadingViewSystem));
            }
            catch (Exception ex)
            {
                EvoDebug.LogWarning($"LoadingSceneSystem failed to unload LoadingScene. {ex.Message}", nameof(LoadingViewSystem));
            }
            finally
            {
                ResetLoadingHandle(release: false);
                if (lockTaken)
                {
                    _transitionLock.Release();
                }
            }

            _loadingProgress?.NotifyHidden();
        }

        private static LoadingScreenView FindLoadingView(Scene scene)
        {
            if (!scene.IsValid())
            {
                return null;
            }

            var roots = scene.GetRootGameObjects();
            for (var i = 0; i < roots.Length; i++)
            {
                var view = roots[i].GetComponentInChildren<LoadingScreenView>(true);
                if (view != null)
                {
                    return view;
                }
            }

            return null;
        }

        private async UniTask WaitMinimumVisibleDurationAsync(CancellationToken cancellationToken)
        {
            var elapsed = Time.realtimeSinceStartupAsDouble - _showStartedAt;
            var delay = MIN_VISIBLE_SECONDS - elapsed;
            if (delay <= 0d)
            {
                return;
            }

            await UniTask.Delay(
                TimeSpan.FromSeconds(delay),
                DelayType.UnscaledDeltaTime,
                PlayerLoopTiming.Update,
                cancellationToken);
        }

        private CancellationTokenSource CreatePresentationTimeout(CancellationToken cancellationToken)
        {
            var source = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken);
            if (_loadingOptions.PresentationTimeoutSeconds > 0f)
            {
                source.CancelAfter(TimeSpan.FromSeconds(_loadingOptions.PresentationTimeoutSeconds));
            }
            return source;
        }

        private void ResetLoadingHandle(bool release)
        {
            if (release && _hasLoadingHandle && _loadingHandle.IsValid())
            {
                _resourceLoader.ReleaseSceneHandle(_loadingScene);
            }

            IsVisible = false;
            _hasLoadingHandle = false;
            _loadingHandle = default;
            _loadingScreenView = null;
        }
    }
}
