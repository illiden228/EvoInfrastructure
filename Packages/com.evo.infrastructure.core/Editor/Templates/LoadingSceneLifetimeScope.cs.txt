using Evo.Infrastructure.Services.UI;
using Cysharp.Threading.Tasks;
using UnityEngine;
using VContainer;
using VContainer.Unity;

namespace Game.Runtime.Loading
{
    public sealed class LoadingSceneLifetimeScope : LifetimeScope
    {
        [SerializeField] private LoadingScreenView view;

        protected override void Configure(IContainerBuilder builder)
        {
            builder.RegisterBuildCallback(container =>
            {
                if (view == null)
                {
                    return;
                }

                var uiService = container.Resolve<IUiService>();
                uiService.RegisterSceneView(view);
                uiService.ShowAsync(view).Forget();
            });
        }
    }
}
